/*	다음과 같이 0~9 사이의 정수를 입력받아 배열에 저장하고, 
	중복 입력된 정수와 입력되지 않은 정수를 출력하고자 한다
	예)
	2 6 3 7 9 5 3 8 5 0
	중복입력된 정수 : 3 5
	입력되지 않은 정수 : 1 4							
	>>조건
	>>조건												*/
#include <iostream>
using namespace std;
#define ARRSIZE 10
int main() 
{
	int iNo[ARRSIZE];
	int iCount[ARRSIZE] = { 0 };
	bool bIsDuplicate = false;
	bool bIsNotInput = false;
	cout << "0~9 사이의 정수 입력 : ";
	// 0~9 사이의 정수를 입력받아 배열에 저장	
	for (int i = 0; i < ARRSIZE; ++i)
	{
		cin >> iNo[i];
		if (iNo[i] >= 0 && iNo[i] <= 9) 
		{
			iCount[iNo[i]]++;
		}
	}
	cout << endl;
	// 중복 입력된 정수 처리 및 출력
	cout << "중복 입력된 정수 : ";
	for (int j = 0; j < ARRSIZE; ++j) 
	{
		if (iCount[j] > 1) 
		{
			cout << j << " ";
			bIsDuplicate = true;
		}
	}
	if (!bIsDuplicate) 
	{
		cout << "중복 입력된 정수 없음";
	}
	cout << endl;
	// 입력되지 않은 정수 처리 및 출력
	cout << "입력되지 않은 정수 : ";
	for (int k = 0; k < ARRSIZE; ++k)
	{
		if (iCount[k] == 0)
		{
			cout << k << " ";
			bIsNotInput = true;
		}
	}	
	if (!bIsNotInput) 
	{
			cout << "0~9 사이에서 입력되지 않은 정수 없음";
		
	}
}
